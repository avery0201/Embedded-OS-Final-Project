{
    HAL_GPIO_WritePin(N_RED_PORT, N_RED_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(S_RED_PORT, S_RED_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(E_RED_PORT, E_RED_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(W_RED_PORT, W_RED_PIN, GPIO_PIN_SET);

    HAL_GPIO_WritePin(N_YELLOW_PORT, N_YELLOW_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(S_YELLOW_PORT, S_YELLOW_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(E_YELLOW_PORT, E_YELLOW_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(W_YELLOW_PORT, W_YELLOW_PIN, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(N_GREEN_PORT, N_GREEN_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(S_GREEN_PORT, S_GREEN_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(E_GREEN_PORT, E_GREEN_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(W_GREEN_PORT, W_GREEN_PIN, GPIO_PIN_RESET);
}

void SetNSGreen(void)
{
    AllLightsRed();

    HAL_GPIO_WritePin(N_RED_PORT, N_RED_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(S_RED_PORT, S_RED_PIN, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(N_GREEN_PORT, N_GREEN_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(S_GREEN_PORT, S_GREEN_PIN, GPIO_PIN_SET);
}

void SetNSYellow(void)
{
    AllLightsRed();

    HAL_GPIO_WritePin(N_RED_PORT, N_RED_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(S_RED_PORT, S_RED_PIN, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(N_YELLOW_PORT, N_YELLOW_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(S_YELLOW_PORT, S_YELLOW_PIN, GPIO_PIN_SET);
}

void SetEWGreen(void)
{
    AllLightsRed();

    HAL_GPIO_WritePin(E_RED_PORT, E_RED_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(W_RED_PORT, W_RED_PIN, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(E_GREEN_PORT, E_GREEN_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(W_GREEN_PORT, W_GREEN_PIN, GPIO_PIN_SET);

void SetEWYellow(void)
{
    AllLightsRed();

    HAL_GPIO_WritePin(E_RED_PORT, E_RED_PIN, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(W_RED_PORT, W_RED_PIN, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(E_YELLOW_PORT, E_YELLOW_PIN, GPIO_PIN_SET);
    HAL_GPIO_WritePin(W_YELLOW_PORT, W_YELLOW_PIN, GPIO_PIN_SET);
}

void StartTrafficTimer(void)
{
    stateElapsedMs = 0;
    __HAL_TIM_SET_COUNTER(&htim2, 0);
    HAL_TIM_Base_Start_IT(&htim2);
}

void ProcessState(void)
{
    stateElapsedMs = 0;

    switch(currentState)
    {
        case STATE_NS_GREEN:
            SetNSGreen();
            break;

        case STATE_NS_YELLOW:
            SetNSYellow();
            break;

        case STATE_ALL_RED_TO_EW:
            AllLightsRed();
            break;

        case STATE_EW_GREEN:
            SetEWGreen();
            break;

        case STATE_EW_YELLOW:
            SetEWYellow();
            break;

        case STATE_ALL_RED_TO_NS:
            AllLightsRed();
            break;

        default:
            currentState = STATE_NS_GREEN;
            SetNSGreen();
            break;
    }
}

/**
  * @brief  Function implementing the TrafficTask thread.
  * @param  argument: Not used
  * @retval None
  */
/* USER CODE END Header_Traffic_Task */
void Traffic_Task(void const * argument)
{
  TrafficEvent event;
  ProcessState();
  StartTrafficTimer();

  for(;;)
  {
    if (xQueueReceive(trafficQueue, &event, portMAX_DELAY) == pdPASS)
    {
      switch(currentState)
      {
        case STATE_NS_GREEN:
          /* Logic: If East-West is waiting (ew_ped_request),
             transition to Yellow after 2 seconds.
          */
          uint32_t ns_limit = (ew_ped_request == 1) ? MIN_GREEN_TIME_MS : GREEN_TIME_MS;

          if (stateElapsedMs >= ns_limit)
          {
            currentState = STATE_NS_YELLOW;
            ProcessState();
          }
          break;

        case STATE_NS_YELLOW:
          if (stateElapsedMs >= YELLOW_TIME_MS)
          {
            ns_ped_request = 0; // Clear NS request as lights turn Red
            currentState = STATE_ALL_RED_TO_EW;
            ProcessState();
          }
          break;

        case STATE_ALL_RED_TO_EW:
          if (stateElapsedMs >= ALL_RED_TIME_MS)
          {
            currentState = STATE_EW_GREEN;
            ProcessState();
          }
          break;

        case STATE_EW_GREEN:
          /* Logic: If North-South is waiting (ns_ped_request),
             transition to Yellow after 2 seconds.
          */
          uint32_t ew_limit = (ns_ped_request == 1) ? MIN_GREEN_TIME_MS : GREEN_TIME_MS;

          if (stateElapsedMs >= ew_limit)
          {
            currentState = STATE_EW_YELLOW;
            ProcessState();
          }
          break;

        case STATE_EW_YELLOW:
          if (stateElapsedMs >= YELLOW_TIME_MS)
          {
            ew_ped_request = 0; // Clear EW request as lights turn Red
            currentState = STATE_ALL_RED_TO_NS;
            ProcessState();
          }
          break;

        case STATE_ALL_RED_TO_NS:
          if (stateElapsedMs >= ALL_RED_TIME_MS)
          {
            currentState = STATE_NS_GREEN;
            ProcessState();
          }
          break;

        default:
          currentState = STATE_NS_GREEN;
          ProcessState();
          break;
      }
    }
  }
}

void BlinkTask(void const * argument)
{
  for(;;)
  {
    /* North-South Blue LED Logic */
    // Only blink if there is a request AND NS is currently Green or Yellow
    if (ns_ped_request == 1 && (currentState == STATE_NS_GREEN || currentState == STATE_NS_YELLOW))
    {
      HAL_GPIO_TogglePin(NE_BLUE_PORT, NE_BLUE_PIN);
    }
    else
    {
      HAL_GPIO_WritePin(NE_BLUE_PORT, NE_BLUE_PIN, GPIO_PIN_RESET);
    }

    /* East-West Blue LED Logic */
    // Only blink if there is a request AND EW is currently Green or Yellow
    if (ew_ped_request == 1 && (currentState == STATE_EW_GREEN || currentState == STATE_EW_YELLOW))
    {
      HAL_GPIO_TogglePin(SW_BLUE_PORT, SW_BLUE_PIN);
    }
    else
    {
      HAL_GPIO_WritePin(SW_BLUE_PORT, SW_BLUE_PIN, GPIO_PIN_RESET);
    }

    osDelay(500); // 500ms toggle = 1 second cycle
  }
}

/**
* @brief Function implementing the PedestrianTask thread.
* @param argument: Not used
*/
void PedestrianTask(void const * argument)
{
  for(;;)
  {
    if(HAL_GPIO_ReadPin(GPIOA, NS_BUTTON_PIN) == GPIO_PIN_SET)
    {
      if (xSemaphoreTake(pedMutex, portMAX_DELAY) == pdTRUE)
      {
        ns_ped_request = 1; // Store request
        xSemaphoreGive(pedMutex);
      }
    }

    if(HAL_GPIO_ReadPin(GPIOA, EW_BUTTON_PIN) == GPIO_PIN_SET)
    {
      if (xSemaphoreTake(pedMutex, portMAX_DELAY) == pdTRUE)
      {
        ew_ped_request = 1; // Store request
        xSemaphoreGive(pedMutex);
      }
    }
    osDelay(50);
  }
}

/**
  * @brief  Period elapsed callback in non blocking mode
  * @note   This function is called  when TIM6 interrupt took place, inside
  * HAL_TIM_IRQHandler(). It makes a direct call to HAL_IncTick() to increment
  * a global variable "uwTick" used as application time base.
  * @param  htim : TIM handle
  * @retval None
  */
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
  if (htim->Instance == TIM6)
  {
    HAL_IncTick();
  }

  /* USER CODE BEGIN Callback 1 */
  // Add this block for your Traffic Timer
  if (htim->Instance == TIM2)
  {
    stateElapsedMs++; // Increment the stopwatch

    TrafficEvent event = EVENT_TIMER_TICK;
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;

    // Send the tick event to the queue to wake up the TrafficTask
    if (trafficQueue != NULL) {
      xQueueSendFromISR(trafficQueue, &event, &xHigherPriorityTaskWoken);
    }

    // This ensures the task runs immediately if it's the highest priority
    portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
  }
  /* USER CODE END Callback 1 */
}
